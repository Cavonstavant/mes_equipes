543 static inline bool memset_server_response_data(
544 server_response_t *data,
545 int code)
546 {
547  server_response_t template;
548  int i = 0;
549 
550  if (!data)
551  return false;
552  while (i < 24 && server_response_template[i].code != code)
553  i++;
554  if (i == 24)
555  return false;
556  template = server_response_template[i];
557  data->callback = template.callback;
558  data->code = template.code;
559  data->data = template.data;
560  return true;
561 }
